sudo yum update
sudo yum upgrade

sudo tar -xvzf jdk-22.0.1_linux-x64_bin.tar.gz -C /opt/

sudo nano /etc/profile

export JAVA_HOME=/opt/jdk-22.0.1
export PATH=$PATH:$JAVA_HOME/bin

source /etc/profile

java -version
javac -version

sudo update-alternatives --install /usr/bin/java java /opt/jdk-22.0.1/bin/java 1
sudo update-alternatives --install /usr/bin/javac javac /opt/jdk-22.0.1/bin/javac 1

sudo update-alternatives --config java
sudo update-alternatives --config javac
